\begin{quote}
\emph{The \emph{Taylor series} of $f(x)$ at basepoint $a$ is}
\begin{align*}
   f(x) &= \sum_{n=0}^\infty \frac{f^{(n)}(a)}{n!} (x-a)^n \\
        &= f(a) + f'(a)(x-a) + \frac{f''(a)}{2}(x-a)^2 + \frac{f'''(a)}{3!}(x-a)^3 + \dots.
\end{align*}
(\emph{When $a=0$ one calls it a \emph{Maclaurin series}, but who cares really?})  \emph{The \emph{nth Taylor polynomial} is the partial sum of the series:}
  $$p_n(x) = f(a) + f'(a)(x-a) + \frac{f''(a)}{2}(x-a)^2 + \dots + \frac{f^{(n)}(a)}{n!}(x-a)^n$$
\end{quote}
