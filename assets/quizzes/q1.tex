30 minutes maximum.  24 points possible; each part is worth 2 points.  No aids (book, calculator, etc.) are permitted.  Show all work and use proper notation for full credit.  Answers should be in reasonably-simplified form.

\problem{12}  Compute the derivatives of the following functions.

\begin{subproblems}
% e and prod/quotient
\item  $\ds f(x)=\frac{e^x}{x^3}$
\vfill

% double chain, ln
\item   $\ds f(x) = \left( \ln(x^2+e^2)\right)^5$
\vfill

%nonstandard base
%exponential + chain
\item   $\ds f(x)=a^{\sin(x)}$ where $a$ is a constant, $a >1$
\vfill

\newpage
%secant
%quotient in chain
\item   $\ds f(x) =\sec\left(\frac{x}{x+1} \right)$
\vfill

\item  $\ds f(x) = e^{\pi x+1}+\sqrt{3}\tan(\pi x)$
\vfill

\item  Find $\ds \frac{dy}{dx}$ if $\ds 2x+y=\cos(xy)$.  (You must solve for $\ds \frac{dy}{dx}$.)
\vfill
\end{subproblems}


\newpage
\problem{12}  Compute the following definite integrals and antiderivatives (indefinite integrals).  Remember that antiderivatives need a ``$+C$''.

\begin{subproblems}
\item $\ds \int_1^2 \frac{2+x^3}{x^2} \; dx$
\vfill

\item $\ds \int \frac{e^{3x}}{\sqrt{5+e^{3x}}}\; dx$
\vfill

\item $\ds \int \frac{1}{x} + \sec(x)\tan(x) \; dx$
\vfill

\newpage
\item $\ds \int x\sqrt{2-x} \; dx$
\vfill

\item $\ds \int_0^2 e^x\cos(1+e^x)\; dx$
\vfill

\item $\ds \int \tan(x)\sec^2(x)\; dx$
\vfill
\end{subproblems}
