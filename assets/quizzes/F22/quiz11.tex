\problem{8}  Consider the parametric curve
   $$x(t)=5 \cos t, \qquad y(t)=\sin t$$

\begin{subproblems}
\item Determine the slope and the equation of the tangent line at  $t=\pi/2$.
\vspace{3.0in}

$\ds (\text{slope}) = \boxed{\phantom{\begin{matrix} lakjsd alsfjd \\ sdfa \\ sdaf \end{matrix}}} \qquad \text{equation:} \quad \boxed{\phantom{\begin{matrix} lakjsd alsfjd saf sd fad a sdfad ad asfd asdf adsf sdfasdf \\ sdfa \\ sdaf \end{matrix}}}$
\item Eliminate the parameter $t$ to write the curve in rectangular form.
\vfill
\end{subproblems}

\problem{5}  Fully set up, but do not evaluate, an integral for the length of the spiral curve $x(t)=t \cos t$, $y(t)=t\sin t$ from $t=0$ to $t=2\pi$.
\vfill

\problem{5}  Find $\ds \frac{d^2 y}{dx^2}$:
    $$x=t^2-t, \qquad y=t+e^t$$
\vfill

\problem{7}  Find the area under one hump of the cycloid
    $$x(t) = 2 (t - \sin t), \quad y(t) = 2 (1 - \cos t)$$
(\emph{Hint.}  One hump goes from $t=0$ to the next $t$ where $y(t)=0$.)
\vfill
