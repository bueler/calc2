\problem{9}  For each part below, \emph{completely set up}, \emph{but do not evaluate}, an integral for the quantity.

\begin{subproblems}
\item The length of the curve {\large \,$y=\frac{x^2}{8} - \ln x$\,} on the interval $1\le x \le 3$.
\vfill

\item The area of the surface formed by revolving the graph of {\large\,$y=x^4$}, \,on the interval $[-1,1]$, around the $x$-axis.
\vfill

\item The area between the graphs of {\large\,$y=e^{-x^2}$} and {\large\,$y=2x+7$} on the interval $[-1,1]$.
\vfill
\end{subproblems}
